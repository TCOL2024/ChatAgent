    <button onclick="sendMessage()">Senden</button>

    <p id="response"></p>


    <script>

        async function sendMessage() {

            const userInput = document.getElementById("userInput").value;

            const responseElement = document.getElementById("response");

            responseElement.innerHTML = "Lädt...";

            

            try {

                const webhookUrl = „https://hook.us2.make.com/28v27trtikh287mce0f63qb39a7t65pn";

                const response = await fetch(webhookUrl + "?message=" + encodeURIComponent(userInput));

                const data = await response.json();

                responseElement.innerHTML = "Antwort: " + data.response;

            } catch (error) {

                responseElement.innerHTML = "Fehler beim Abrufen der Antwort.";

            }

        }

    </script>
